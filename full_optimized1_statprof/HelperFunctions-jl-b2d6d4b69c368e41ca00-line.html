81 function distribute_params(rates)
82 # 1: initialConc ~ lognormal(log(stdConc), sqrt(log(CV^2)))
83 # 2: scale (no change)
84 # 3-16: rates ~ normal(stdRates, stdRates * CV) ~ stdRates * normal(1, CV)
85
86 New_rates = similar(rates);
87 @inbounds for i in 1:TOTAL_SPECIES
88 # 1: initial concentrations
89 if FOUNDER_CONC_CV > 0
90 mu = log(rates[i, INITIALCONC]);
91 sigma = sqrt(log(FOUNDER_CONC_CV*FOUNDER_CONC_CV+1));
92 New_rates[i, INITIALCONC] = rand(LogNormal(mu, sigma));
93 else
94 New_rates[i, INITIALCONC] = rates[i, INITIALCONC];
95 end
96
97 # 2: scale
98 New_rates[i, SCALE] = rates[i, SCALE];
99
100 # 3-end: rate constants (synthesis, decay, etc.)
101 New_rates[i, 3:end] .= rates[i, 3:end] .* [max(0, rand(Normal(1, FOUNDER_RATE_CV))) for i in 1:(TOTAL_PARAMS-2)];
