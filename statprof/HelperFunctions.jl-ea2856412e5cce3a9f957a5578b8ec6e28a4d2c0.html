125 function distribute_params(rates)
126 # 1: initialConc ~ lognormal(log(stdConc), sqrt(log(CV^2)))
127 # 2: scale (no change)
128 # 3-16: rates ~ normal(stdRates, stdRates * CV) ~ stdRates * normal(1, CV)
129
130 New_rates = similar(rates);
131 @inbounds for i in 1:TOTAL_SPECIES
132 # 1: initial concentrations
133 if FOUNDER_CONC_CV > 0
134 mu = log(rates[i, INITIALCONC]);
135 sigma = sqrt(log(FOUNDER_CONC_CV*FOUNDER_CONC_CV+1));
136 New_rates[i, INITIALCONC] = rand(LogNormal(mu, sigma));
137 else
138 New_rates[i, INITIALCONC] = rates[i, INITIALCONC];
139 end
140
141 # 2: scale
142 New_rates[i, SCALE] = rates[i, SCALE];
143
144 # 3-end: rate constants (synthesis, decay, etc.)
145 New_rates[i, 3:end] .= rates[i, 3:end] .* [max(0, rand(Normal(1, FOUNDER_RATE_CV))) for i in 1:(TOTAL_PARAMS-2)];
146 end
