66 @inline function check_gridded(itpflag, knots, axs)
67 flag, ax1, k1 = getfirst(itpflag), axs[1], knots[1]
68 if flag isa NoInterp
69 k1 == ax1 || error("for NoInterp knot vector should be $ax1, got $k1")
70 else
71 axes(k1, 1) == ax1 || throw(DimensionMismatch("knot vectors must have the same axes as the corresponding dimension of the array"))
72 end
73 degree(flag) isa Union{NoInterp,Constant,Linear} || error("only Linear, Constant, and NoInterp supported, got $flag")
74 length(k1) == 1 && error("dimensions of length 1 not yet supported") # FIXME
75 if issorted(k1)
76 if !allunique(k1)
77 @warn "Duplicated knots were deduplicated. Use Interpolations.deduplicate_knots!(knots) explicitly to avoid this warning." k1
78 deduplicate_knots!(k1)
79 end
80 else
81 error("knot-vectors must be unique and sorted in increasing order")
82 end

88 """
89 Interpolations.deduplicate_knots!(knots)
90
91 Makes knots unique by incrementing repeated but otherwise sorted knots using `nextfloat`.
92
93 # Example
94
95 ```jldoctest
96 julia> knots = [-8.0, 0.0, 20.0, 20.0]
97 4-element Vector{Float64}:
98 -8.0
99 0.0
100 20.0
101 20.0
102
103 julia> Interpolations.deduplicate_knots!(knots)
104 4-element Vector{Float64}:
105 -8.0
106 0.0
107 20.0
108 20.000000000000004
109 ```
110 """
111 function deduplicate_knots!(knots)
112 last_knot = first(knots)
113 for i = eachindex(knots)
114 if i == 1
115 continue
116 end
117 if knots[i] == last_knot
118 @inbounds knots[i] = nextfloat(knots[i-1])
119 else
120 last_knot = @inbounds knots[i]
121 end
122 end
123 knots
124 end
