26 # Standard integral (I) step size controller
27 """
28 IController()
29
30 The standard (integral) controller is the most basic step size controller.
31 This controller is usually the first one introduced in numerical analysis classes
32 but should only be used rarely in practice because of efficiency problems for
33 many problems/algorithms.
34
35 Construct an integral (I) step size controller adapting the time step
36 based on the formula
37 ```
38 Δtₙ₊₁ = εₙ₊₁^(1/k) * Δtₙ
39 ```
40 where `k = get_current_adaptive_order(alg, integrator.cache) + 1` and `εᵢ` is the
41 inverse of the error estimate `integrator.EEst` scaled by the tolerance
42 (Hairer, Nørsett, Wanner, 2008, Section II.4).
43 The step size factor is multiplied by the safety factor `gamma` and clipped to
44 the interval `[qmin, qmax]`.
45 A step will be accepted whenever the estimated error `integrator.EEst` is
46 less than or equal to unity. Otherwise, the step is rejected and re-tried with
47 the predicted step size.
48
49 ## References
50 - Hairer, Nørsett, Wanner (2008)
51 Solving Ordinary Differential Equations I Nonstiff Problems
52 [DOI: 10.1007/978-3-540-78862-1](https://doi.org/10.1007/978-3-540-78862-1)
53 """

55 end

170 # PID step size controller
171 """
172 PIDController(beta1, beta2, beta3=zero(beta1);
173 limiter=default_dt_factor_limiter,
174 accept_safety=0.81)
175
176 The proportional-integral-derivative (PID) controller is a generalization of the
177 [`PIController`](@ref) and can have improved stability and efficiency properties.
178
179 Construct a PID step size controller adapting the time step based on the formula
180 ```
181 Δtₙ₊₁ = εₙ₊₁^(β₁/k) * εₙ^(β₂/k) * εₙ₋₁^(β₃/ k) * Δtₙ
182 ```
183 where `k = min(alg_order, alg_adaptive_order) + 1` and `εᵢ` are inverses of
184 the error estimates scaled by the tolerance (Söderlind, 2003).
185 The step size factor is limited by the `limiter` with default value
186 ```
187 limiter(x) = one(x) + atan(x - one(x))
188 ````
189 as proposed by Söderlind and Wang (2006). A step will be accepted whenever the
190 predicted step size change is bigger than `accept_safety`. Otherwise, the step
191 is rejected and re-tried with the predicted step size.
192
193 Some standard controller parameters suggested in the literature are
194
195 | Controller | `beta1` | `beta2` | `beta3` |
196 |:-----------|--------:|--------:|:-------:|
197 | basic | `1.00` | `0.00` | `0` |
198 | PI42 | `0.60` | `-0.20` | `0` |
199 | PI33 | `2//3` | `-1//3` | `0` |
200 | PI34 | `0.70` | `-0.40` | `0` |
201 | H211PI | `1//6` | `1//6` | `0` |
202 | H312PID | `1//18` | `1//9` | `1//18` |
203
204 !!! note
205 In contrast to the [`PIController`](@ref), the coefficients `beta1, beta2, beta3`
206 are scaled by the order of the method. Thus, standard controllers such as PI42
207 can use the same coefficients `beta1, beta2, beta3` for different algorithms.
208
209 !!! note
210 In contrast to other controllers, the `PIDController` does not use the keyword
211 arguments `qmin, qmax` to limit the step size change or the safety factor `gamma`.
212 These common keyword arguments are replaced by the `limiter` and `accept_safety`
213 to guarantee a smooth behavior (Söderlind and Wang, 2006).
214 Because of this, a `PIDController` behaves different from a [`PIController`](@ref),
215 even if `beta1, beta2` are adapted accordingly and `iszero(beta3)`.
216
217 ## References
218 - Söderlind (2003)
219 Digital Filters in Adaptive Time-Stepping
220 [DOI: 10.1145/641876.641877](https://doi.org/10.1145/641876.641877)
221 - Söderlind, Wang (2006)
222 Adaptive time-stepping and computational stability
223 [DOI: 10.1016/j.cam.2005.03.008](https://doi.org/10.1016/j.cam.2005.03.008)
224 - Ranocha, Dalcin, Parsani, Ketcheson (2021)
225 Optimized Runge-Kutta Methods with Automatic Step Size Control for
226 Compressible Computational Fluid Dynamics
227 [arXiv:2104.06836](https://arxiv.org/abs/2104.06836)
228 """
229 struct PIDController{QT, Limiter} <: AbstractController
230 beta::MVector{3,QT} # controller coefficients
231 err ::MVector{3,QT} # history of the error estimates
232 accept_safety::QT # accept a step if the predicted change of the step size
233 # is bigger than this parameter
234 limiter::Limiter # limiter of the dt factor (before clipping)
235 end

254 @inline function stepsize_controller!(integrator, controller::PIDController, alg)
255 @unpack qmax = integrator.opts
256 beta1, beta2, beta3 = controller.beta
257 controller.err[1] = inv(DiffEqBase.value(integrator.EEst))
258 err1, err2, err3 = controller.err
259
260 iszero(err1) && return qmax
261
262 k = min(alg_order(alg), alg_adaptive_order(alg)) + 1
263 dt_factor = err1^(beta1 / k) * err2^(beta2 / k) * err3^(beta3 / k)
264 dt_factor = controller.limiter(dt_factor)
265 # Note: No additional limiting of the form
266 # dt_factor = max(qmin, min(qmax, dt_factor))
267 # is necessary since the `limiter` should take care of that. The default limiter
268 # ensures
269 # 0.21 ≈ limiter(0) <= dt_factor <= limiter(Inf) ≈ 2.57
270 # See Söderlind, Wang (2006), Section 6.
271 integrator.qold = dt_factor
272 return dt_factor
273 end

297 # Gustafsson predictive step size controller
298 """
299 PredictiveController()
300
301 The Gustafsson acceleration algorithm accelerates changes so that way algorithms
302 can more swiftly change to handle quick transients. This algorithm is thus
303 well-suited for stiff solvers where this can be expected, and is the default
304 for algorithms like the (E)SDIRK methods.
305
306 ```julia
307 gamma = integrator.opts.gamma
308 niters = integrator.cache.newton_iters
309 fac = min(gamma,(1+2*integrator.alg.max_newton_iter)*gamma/(niters+2*integrator.alg.max_newton_iter))
310 expo = 1/(alg_order(integrator.alg)+1)
311 qtmp = (integrator.EEst^expo)/fac
312 @fastmath q = max(inv(integrator.opts.qmax),min(inv(integrator.opts.qmin),qtmp))
313 if q <= integrator.opts.qsteady_max && q >= integrator.opts.qsteady_min
314 q = one(q)
315 end
316 integrator.qold = q
317 q
318 ```
319 In this case, `niters` is the number of Newton iterations which was required
320 in the most recent step of the algorithm. Note that these values are used
321 differently depending on acceptance and rejectance. When the step is accepted,
322 the following logic is applied:
323 ```julia
324 if integrator.success_iter > 0
325 expo = 1/(alg_adaptive_order(integrator.alg)+1)
326 qgus=(integrator.dtacc/integrator.dt)*(((integrator.EEst^2)/integrator.erracc)^expo)
327 qgus = max(inv(integrator.opts.qmax),min(inv(integrator.opts.qmin),qgus/integrator.opts.gamma))
328 qacc=max(q,qgus)
329 else
330 qacc = q
331 end
332 integrator.dtacc = integrator.dt
333 integrator.erracc = max(1e-2,integrator.EEst)
334 integrator.dt/qacc
335 ```
336 When it rejects, its the same as the [`IController`](@ref):
337 ```julia
338 if integrator.success_iter == 0
339 integrator.dt *= 0.1
340 else
341 integrator.dt = integrator.dt/integrator.qold
342 end
343 ```
344 """
345 struct PredictiveController <: AbstractController
346 end

607 # Update gamma and beta1
608 controller.beta1 = typeof(controller.beta1)(1 // (2integrator.cache.n_curr + 1))
609 integrator.opts.gamma = DiffEqBase.fastpow(typeof(integrator.opts.gamma)(1 // 4),controller.beta1)
610 # Compute new stepsize scaling
611 qtmp = EEst * DiffEqBase.fastpow(DiffEqBase.fastpow(tol,(1.0 - s_curr / s_new)),controller.beta1) / integrator.opts.gamma
612 @fastmath q = max(inv(integrator.opts.qmax),min(inv(integrator.opts.qmin),qtmp))
613 end
614 integrator.cache.Q[n_new - alg.n_min + 1] = q
615 end

617 function step_accept_controller!(integrator,alg::Union{ExtrapolationMidpointDeuflhard,ImplicitDeuflhardExtrapolation},q)
618 # Compute new order and stepsize, return new stepsize
619 @unpack n_min, n_max = alg
620 @unpack n_curr, n_old, Q = integrator.cache
621 s = integrator.cache.stage_number
622
623 # Compute new order based on available quantities
624 tmp = (n_min:n_curr) .- n_min .+ 1 # Index range of quantities computed so far
625 dt_new = Vector{eltype(Q)}(undef,length(tmp)+1)
626 dt_new[1:end-1] = integrator.dt ./ Q[tmp] # Store for the possible new stepsizes
627 dtmin = timedepentdtmin(integrator)
628 dt_new[1:end-1] = max.(dtmin, min.(abs(integrator.opts.dtmax), abs.(dt_new[1:end-1]))) # Safety scaling
629
630 # n_new is the most efficient order of the last step
631 work = s[tmp] ./ dt_new[1:end-1]
632 n_new = argmin(work) + n_min - 1
633
634 # Check if n_new may be increased
635 if n_new == n_curr < min(n_max, n_old + 1) # cf. win_max in perfom_step! of the last step
636 # Predict stepsize scaling for order (n_new + 1)
637 stepsize_predictor!(integrator, alg, n_new+1) # Update cache.Q
638
639 # Compute and scale the corresponding stepsize
640 dt_new[end] = integrator.dt ./ Q[tmp[end]+1]
641 dt_new[end] = max(dtmin, min(abs(integrator.opts.dtmax), abs.(dt_new[end])))
642
643 # Check if (n_new + 1) would have been more efficient than n_new
644 if work[end] > s[tmp[end]+1] / dt_new[end]
645 n_new = n_new + 1
646 end
647 end
648
649 integrator.cache.n_curr = n_new
650 dt_new[n_new - n_min + 1]
651 end

653 function step_reject_controller!(integrator, alg::Union{ExtrapolationMidpointDeuflhard,ImplicitDeuflhardExtrapolation})
654 # Compute and save reduced stepsize dt_red of order n_old
655 # Use the latest error estimate to predict dt_red if an estimate of order n_old is not available
656 if integrator.cache.n_curr < integrator.cache.n_old
657 stepsize_predictor!(integrator,alg,integrator.cache.n_old) # Update cache.Q
658 end
659 integrator.cache.n_curr = integrator.cache.n_old # Reset order for redoing the rejected step
660 dt_red = integrator.dt / integrator.cache.Q[integrator.cache.n_old - integrator.alg.n_min + 1]
661 dtmin = timedepentdtmin(integrator)
662 dt_red = integrator.tdir*max(dtmin, min(abs(integrator.opts.dtmax), abs(dt_red))) # Safety scaling
663 integrator.dt = dt_red
664 end

694 end

696 else
697 if iszero(integrator.EEst)

699 else

702 integrator.opts.gamma = DiffEqBase.fastpow(typeof(integrator.opts.gamma)(65 // 100),controller.beta1)

711 function step_accept_controller!(integrator,alg::Union{ExtrapolationMidpointHairerWanner, ImplicitHairerWannerExtrapolation, ImplicitEulerExtrapolation, ImplicitEulerBarycentricExtrapolation},q)
712 # Compute new order and stepsize, return new stepsize
713 @unpack n_min, n_max = alg
714 @unpack n_curr, n_old, Q, sigma, work, dt_new = integrator.cache
715 s = integrator.cache.stage_number
716
717 # Compute new order based on available quantities
718 win_min_old = min(n_old, n_curr) - 1 # cf. win_min in perfom_step! of the last step
719 tmp = win_min_old:(max(n_curr, n_old) + 1) # Index range for the new order
720 #@show size(dt_new)
721 fill!(dt_new, zero(eltype(dt_new)))
722 @.. Q = integrator.dt/Q
723 copyto!(dt_new,win_min_old,Q,win_min_old,(max(n_curr, n_old) + 1) - win_min_old + 1)
724 @.. Q = integrator.dt/Q
725 dtmin = timedepentdtmin(integrator)
726 fill!(work,zero(eltype(work))) # work[n] is the work for order (n-1)
727 for i in tmp
728 work[i] = s[i]/dt_new[i]
729 end
730 # Order selection
731 n_new = n_old
732 if n_curr == n_min # Enforce n_min + 1 ≦ n_new
733 n_new = n_min + 1
734 else
735 if n_curr <= n_old
736 if work[n_curr-1] < sigma * work[n_curr]
737 n_new = max(n_curr-1,n_old-1,n_min+1) # Enforce n_min + 1≦ n_new
738 elseif work[n_curr] < sigma * work[n_curr-1]
739 n_new = min(n_curr+1,n_max-1) # Enforce n_new ≦ n_max - 1
740 else
741 n_new = n_curr # n_min + 1 ≦ n_curr
742 end
743 else
744 if work[n_old] < sigma * work[n_old+1]
745 n_new = max(n_old-1,n_min+1) # Enforce n_min + 1 ≦ n_new
746 end
747 if work[n_curr+1] < sigma * work[n_new+1]
748 n_new = min(n_new+1,n_max-1) # Enforce n_new ≦ n_max - 1
749 end
750 end
751 end
752 integrator.cache.n_curr = n_new
753
754 # Stepsize selection
755 if n_new == n_curr + 1
756 # Compute the new stepsize of order n_new based on the optimal stepsize of order n_curr
757 dt_new[n_new+1] = s[n_curr + 2]/s[n_curr + 1 ] * dt_new[n_curr+1]
758 dt_new[n_new+1] = max(dtmin, min(abs(integrator.opts.dtmax), abs(dt_new[n_new+1])))
759 end
760 dt_new[n_new + 1]
761 end

763 function step_reject_controller!(integrator, alg::Union{ExtrapolationMidpointHairerWanner, ImplicitHairerWannerExtrapolation, ImplicitEulerExtrapolation, ImplicitEulerBarycentricExtrapolation})
764 # Compute and save order and stepsize for redoing the current step
765 @unpack n_old, n_curr, Q = integrator.cache
766
767 # Order selection
768 n_red = n_old
769 if n_curr == n_old - 1
770 n_red = max(alg.n_min+1,n_old-1) # Enforce n_min + 1 ≦ n_red
771 end
772 integrator.cache.n_curr = n_red
773
774 # Stepsize selection
775 dt_red = integrator.dt / Q[n_red + 1]
776 dtmin = timedepentdtmin(integrator)
777 dt_red = integrator.tdir*max(dtmin, min(abs(integrator.opts.dtmax), abs(dt_red))) # Safety scaling
778 integrator.dt = dt_red
779 end
