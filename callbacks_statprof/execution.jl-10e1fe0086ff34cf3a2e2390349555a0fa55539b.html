175 ####################
176 # parameter tuning #
177 ####################
178
179 # The tuning process is as follows:
180 #
181 # 1. Using `lineartrial`, take one sample of the benchmark for each `evals` in `1:RESOLUTION`.
182 #
183 # 2. Extract the minimum sample found in this trial. Hopefully, this value will be
184 # reasonably close to the true benchmark time. At the very least, we can be certain
185 # that the minimum sample overcomes discretization noise, assuming a sufficient time
186 # budget for the trial (which technically should never need to be more than a couple of
187 # seconds if `RESOLUTION` âˆ¼ 1Î¼s).
188 #
189 # 3. Using the sample from step 2 as a reasonable estimate `t`, calculate `evals`. This
190 # is done by indexing into a pregenerated table that maps potential `t` values to the
191 # appropriate `evals` values. The actual generation of this table is accomplished via
192 # heuristic, driven mainly by empircal insight. It's built on the following postulates:
193 # - Small variations in `t` should not cause large variations in `evals`
194 # - `RESOLUTION` <= 1000ns
195 # - `evals` <= `RESOLUTION` if `RESOLUTION` is in nanoseconds and `t` >= 1ns
196 # - Increasing `t` should generally decrease `evals`
197 #
198 # 4. By default, tune the `samples` parameter as a function of `t`, `evals`, and the time
199 # budget for the benchmark.
200
201 # the logistic function is useful for determining `evals` for `1 < t < RESOLUTION`
202 logistic(u, l, k, t, t0) = round(Int, ((u - l) / (1 + exp(-k * (t - t0)))) + l)
203
204 const EVALS = Vector{Int}(undef, 9000) # any `t > length(EVALS)` should get an `evals` of 1
205 for t in 1:400 (EVALS[t] = logistic(1006, 195, -0.025, t, 200)) end # EVALS[1] == 1000, EVALS[400] == 200
206 for t in 401:1000 (EVALS[t] = logistic(204, -16, -0.01, t, 800)) end # EVALS[401] == 200, EVALS[1000] == 10
207 for i in 1:8 (EVALS[((i*1000)+1):((i+1)*1000)] .= 11 - i) end # linearly decrease from EVALS[1000]
208
209 guessevals(t) = t <= length(EVALS) ? EVALS[t] : 1

573 return esc(quote
574 local $bench = $BenchmarkTools.@benchmarkable $(args...)
575 $BenchmarkTools.warmup($bench)
576 $tune_phase
577 local $trial, $result = $BenchmarkTools.run_result($bench)
578 local $trialmin = $BenchmarkTools.minimum($trial)
579 local $trialallocs = $BenchmarkTools.allocs($trialmin)
580 println(" ",
581 $BenchmarkTools.prettytime($BenchmarkTools.time($trialmin)),
582 " (", $trialallocs , " allocation",
583 $trialallocs == 1 ? "" : "s", ": ",
584 $BenchmarkTools.prettymemory($BenchmarkTools.memory($trialmin)), ")")
585 $result
586 end)
587 end

591
592 Run `@benchmark` while profiling. This is similar to
593
594 @profile @benchmark expression [other parameters...]
595
596 but the profiling is applied only to the main
597 execution (after compilation and tuning).
598 The profile buffer is cleared prior to execution.
599
600 View the profile results with `Profile.print(...)`.
601 See the profiling section of the Julia manual for more
602 information.
603 """
604 macro bprofile(args...)
605 _, params = prunekwargs(args...)
606 if !haskw(args, :gctrial)
607 args = (args..., Expr(:kw, :gctrial, false))
608 end
609 if !haskw(args, :gcsample)
610 args = (args..., Expr(:kw, :gcsample, false))
611 end
612 tmp = gensym()
613 return esc(quote
614 local $tmp = $BenchmarkTools.@benchmarkable $(args...)
615 $BenchmarkTools.warmup($tmp)
616 $(hasevals(params) ? :() : :($BenchmarkTools.tune!($tmp)))
617 $BenchmarkTools.Profile.clear()
618 $BenchmarkTools.@profile $BenchmarkTools.run($tmp)
619 end)
