554 This macro requires Julia 1.7 or later.
555 """
556 macro invoke(ex)
557 f, args, kwargs = destructure_callex(ex)
558 arg2typs = map(args) do x
559 is_expr(x, :(::)) ? (x.args...,) : (x, GlobalRef(Core, :Any))
560 end
561 args, argtypes = first.(arg2typs), last.(arg2typs)
562 return esc(:($(GlobalRef(Core, :invoke))($(f), Tuple{$(argtypes...)}, $(args...); $(kwargs...))))
563 end

565 """
566 @invokelatest f(args...; kwargs...)
567 Provides a convenient way to call [`Base.invokelatest`](@ref).
568 `@invokelatest f(args...; kwargs...)` will simply be expanded into
569 `Base.invokelatest(f, args...; kwargs...)`.
570
571 !!! compat "Julia 1.7"
572 This macro requires Julia 1.7 or later.
573 """
574 macro invokelatest(ex)
575 f, args, kwargs = destructure_callex(ex)
576 return esc(:($(GlobalRef(Base, :invokelatest))($(f), $(args...); $(kwargs...))))
577 end
578
579 function destructure_callex(ex)
580 is_expr(ex, :call) || throw(ArgumentError("a call expression f(args...; kwargs...) should be given"))
581
582 f = first(ex.args)
583 args = []
584 kwargs = []
585 for x in ex.args[2:end]
586 if is_expr(x, :parameters)
587 append!(kwargs, x.args)
588 elseif is_expr(x, :kw)
589 push!(kwargs, x)
590 else
591 push!(args, x)
592 end
593 end
594
595 return f, args, kwargs
596 end

600 """
601 Base.runtests(tests=["all"]; ncores=ceil(Int, Sys.CPU_THREADS / 2),
602 exit_on_error=false, revise=false, [seed])
603
604 Run the Julia unit tests listed in `tests`, which can be either a string or an array of
605 strings, using `ncores` processors. If `exit_on_error` is `false`, when one test
606 fails, all remaining tests in other files will still be run; they are otherwise discarded,
607 when `exit_on_error == true`.
608 If `revise` is `true`, the `Revise` package is used to load any modifications to `Base` or
609 to the standard libraries before running the tests.
610 If a seed is provided via the keyword argument, it is used to seed the
611 global RNG in the context where the tests are run; otherwise the seed is chosen randomly.
612 """
613 function runtests(tests = ["all"]; ncores::Int = ceil(Int, Sys.CPU_THREADS::Int / 2),
614 exit_on_error::Bool=false,
615 revise::Bool=false,
616 seed::Union{BitInteger,Nothing}=nothing)
617 if isa(tests,AbstractString)
618 tests = split(tests)
619 end
620 exit_on_error && push!(tests, "--exit-on-error")
621 revise && push!(tests, "--revise")
622 seed !== nothing && push!(tests, "--seed=0x$(string(seed % UInt128, base=16))") # cast to UInt128 to avoid a minus sign
623 ENV2 = copy(ENV)
624 ENV2["JULIA_CPU_THREADS"] = "$ncores"
625 ENV2["JULIA_DEPOT_PATH"] = mktempdir(; cleanup = true)
626 delete!(ENV2, "JULIA_LOAD_PATH")
627 delete!(ENV2, "JULIA_PROJECT")
628 try
629 run(setenv(`$(julia_cmd()) $(joinpath(Sys.BINDIR::String,
630 Base.DATAROOTDIR, "julia", "test", "runtests.jl")) $tests`, ENV2))
631 nothing
632 catch
633 buf = PipeBuffer()
634 Base.require(Base, :InteractiveUtils).versioninfo(buf)
635 error("A test has failed. Please submit a bug report (https://github.com/JuliaLang/julia/issues)\n" *
636 "including error messages above and the output of versioninfo():\n$(read(buf, String))")
637 end
638 end
