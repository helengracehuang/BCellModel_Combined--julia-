21 # Wrapper for an OS file descriptor (on both Unix and Windows)
22 """
23 RawFD
24
25 Primitive type which wraps the native OS file descriptor.
26 `RawFD`s can be passed to methods like [`stat`](@ref) to
27 discover information about the underlying file, and can
28 also be used to open streams, with the `RawFD` describing
29 the OS file backing the stream.
30 """
31 primitive type RawFD 32 end
32 RawFD(fd::Integer) = bitcast(RawFD, Cint(fd))
33 RawFD(fd::RawFD) = fd
34 Base.cconvert(::Type{Cint}, fd::RawFD) = bitcast(Cint, fd)
35
36 dup(x::RawFD) = ccall((@static Sys.iswindows() ? :_dup : :dup), RawFD, (RawFD,), x)
37 dup(src::RawFD, target::RawFD) = systemerror("dup", -1 ==
38 ccall((@static Sys.iswindows() ? :_dup2 : :dup2), Int32,
39 (RawFD, RawFD), src, target))

52 function dup(src::WindowsRawSocket)
53 new_handle = Ref(WindowsRawSocket(Ptr{Cvoid}(-1)))

60 return new_handle[]
61 end

199 n == 0 && return ""

203 """
204 strptime([format], timestr)
205
206 Parse a formatted time string into a `TmStruct` giving the seconds, minute, hour, date, etc.
207 Supported formats are the same as those in the standard C library. On some platforms,
208 timezones will not be parsed correctly. If the result of this function will be passed to
209 `time` to convert it to seconds since the epoch, the `isdst` field should be filled in
210 manually. Setting it to `-1` will tell the C library to use the current system settings to
211 determine the timezone.
212 """
213 strptime(timestr::AbstractString) = strptime("%c", timestr)
214 function strptime(fmt::AbstractString, timestr::AbstractString)
215 tm = TmStruct()
216 r = ccall(:strptime, Cstring, (Cstring, Cstring, Ref{TmStruct}), timestr, fmt, tm)
217 # the following would tell mktime() that this is a local time, and that
218 # it should try to guess the timezone. not sure if/how this should be
219 # exposed in the API.
220 # tm.isdst = -1
221 if r == C_NULL
222 # TODO: better error message
223 throw(ArgumentError("invalid arguments"))
224 end
225 @static if Sys.isapple()
226 # if we didn't explicitly parse the weekday or year day, use mktime
227 # to fill them in automatically.
228 if !occursin(r"([^%]|^)%(a|A|j|w|Ow)", fmt)
229 ccall(:mktime, Int, (Ref{TmStruct},), tm)
230 end
231 end
232 return tm
233 end

381 """
382 rand([T::Type])
383
384 Interface to the C `rand()` function. If `T` is provided, generate a value of type `T`
385 by composing two calls to `rand()`. `T` can be `UInt32` or `Float64`.
386 """
387 rand() = ccall(:rand, Cint, ())
388 @static if Sys.iswindows()
389 # Windows RAND_MAX is 2^15-1
390 rand(::Type{UInt32}) = ((rand() % UInt32) << 17) ⊻ ((rand() % UInt32) << 8) ⊻ (rand() % UInt32)
391 else
392 # RAND_MAX is at least 2^15-1 in theory, but we assume 2^16-1
393 # on non-Windows systems (in practice, it's 2^31-1)
394 rand(::Type{UInt32}) = ((rand() % UInt32) << 16) ⊻ (rand() % UInt32)
395 end
396 rand(::Type{Float64}) = rand(UInt32) * 2.0^-32

434 function getpwuid(uid::Unsigned, throw_error::Bool=true)
435 ref_pd = Ref(Cpasswd())
436 ret = ccall(:jl_os_get_passwd, Cint, (Ref{Cpasswd}, Culong), ref_pd, uid)
437 if ret != 0
438 throw_error && Base.uv_error("getpwuid", ret)
439 return
440 end
441 pd = ref_pd[]
442 pd = Passwd(
443 pd.username == C_NULL ? "" : unsafe_string(pd.username),
444 pd.uid,
445 pd.gid,
446 pd.shell == C_NULL ? "" : unsafe_string(pd.shell),
447 pd.homedir == C_NULL ? "" : unsafe_string(pd.homedir),
448 pd.gecos == C_NULL ? "" : unsafe_string(pd.gecos),
449 )
450 ccall(:uv_os_free_passwd, Cvoid, (Ref{Cpasswd},), ref_pd)
451 return pd

453 function getgrgid(gid::Unsigned, throw_error::Bool=true)
454 ref_gp = Ref(Cgroup())
455 ret = ccall(:jl_os_get_group, Cint, (Ref{Cgroup}, Culong), ref_gp, gid)
456 if ret != 0
457 throw_error && Base.uv_error("getgrgid", ret)
458 return
459 end
460 gp = ref_gp[]
461 members = String[]
462 if gp.mem != C_NULL
463 while true
464 mem = unsafe_load(gp.mem, length(members) + 1)
465 mem == C_NULL && break
466 push!(members, unsafe_string(mem))
467 end
468 end
469 gp = Group(
470 gp.groupname == C_NULL ? "" : unsafe_string(gp.groupname),
471 gp.gid,
472 members,
473 )
474 ccall(:jl_os_free_group, Cvoid, (Ref{Cgroup},), ref_gp)
475 return gp
476 end
