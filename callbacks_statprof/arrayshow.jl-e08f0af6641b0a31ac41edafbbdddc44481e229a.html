5 # Printing a value requires to take into account the :typeinfo property
6 # from the IO context; this property encodes (as a type) the type information
7 # that is supposed to have already been displayed concerning this value,
8 # so that redundancy can be avoided. For example, when printing an array of
9 # `Float16` values, the header "Float16" will be printed, and the values
10 # can simply be printed with the decimal representations:
11 # show(Float16(1)) -> "Float16(1.0)"
12 # show([Float16(1)]) -> "Float16[1.0]" (instead of "Float16[Float16(1.0)]")
13 # Similarly:
14 # show([[Float16(1)]]) -> "Array{Float16}[[1.0]]" (instead of "Array{Float16}[Float16[1.0]]")
15 #
16 # The array printing methods here can be grouped into two categories (and are annotated as such):
17 # 1) "typeinfo aware" : these are "API boundaries" functions, which will read the typeinfo
18 # property from the context, and pass down to their value an updated property
19 # according to its eltype; at each layer of nesting, only one "typeinfo aware"
20 # function must be called;
21 # 2) "typeinfo agnostic": these are helper functions used by the first category; hence
22 # they don't manipulate the typeinfo property, and let the printing routines
23 # for their elements read directly the property set by their callers
24 #
25 # Non-annotated functions are even lower level (e.g. print_matrix_row), so they fall
26 # by default into category 2.
27 #
28 # The basic organization of this file is
29 # 1) printing with `display`
30 # 2) printing with `show`
31 # 3) Logic for displaying type information

49 """
50 `alignment(io, X, rows, cols, cols_if_complete, cols_otherwise, sep)` returns the
51 alignment for specified parts of array `X`, returning the (left,right) info.
52 It will look in X's `rows`, `cols` (both lists of indices)
53 and figure out what's needed to be fully aligned, for example looking all
54 the way down a column and finding out the maximum size of each element.
55 Parameter `sep::Integer` is number of spaces to put between elements.
56 `cols_if_complete` and `cols_otherwise` indicate screen width to use.
57 Alignment is reported as a vector of (left,right) tuples, one for each
58 column going across the screen.
59 """
60 function alignment(io::IO, @nospecialize(X::AbstractVecOrMat),
61 rows::AbstractVector{T}, cols::AbstractVector{V},
62 cols_if_complete::Integer, cols_otherwise::Integer, sep::Integer) where {T,V}
63 a = Tuple{T, V}[]
64 for j in cols # need to go down each column one at a time
65 l = r = 0
66 for i in rows # plumb down and see what largest element sizes are
67 if isassigned(X,i,j)
68 aij = alignment(io, X[i,j])::Tuple{Int,Int}
69 else
70 aij = undef_ref_alignment
71 end
72 l = max(l, aij[1]) # left characters
73 r = max(r, aij[2]) # right characters
74 end
75 push!(a, (l, r)) # one tuple per column of X, pruned to screen width
76 if length(a) > 1 && sum(map(sum,a)) + sep*length(a) >= cols_if_complete
77 pop!(a) # remove this latest tuple if we're already beyond screen width
78 break
79 end
80 end
81 if 1 < length(a) < length(axes(X,2))
82 while sum(map(sum,a)) + sep*length(a) >= cols_otherwise
83 pop!(a)
84 end
85 end
86 return a
87 end

89 """
90 `print_matrix_row(io, X, A, i, cols, sep)` produces the aligned output for
91 a single matrix row X[i, cols] where the desired list of columns is given.
92 The corresponding alignment A is used, and the separation between elements
93 is specified as string sep.
94 `print_matrix_row` will also respect compact output for elements.
95 """
96 function print_matrix_row(io::IO,
97 @nospecialize(X::AbstractVecOrMat), A::Vector,
98 i::Integer, cols::AbstractVector, sep::AbstractString)
99 for (k, j) = enumerate(cols)
100 k > length(A) && break
101 if isassigned(X,Int(i),Int(j)) # isassigned accepts only `Int` indices
102 x = X[i,j]
103 a = alignment(io, x)::Tuple{Int,Int}
104
105 # First try 3-arg show
106 sx = sprint(show, "text/plain", x, context=io, sizehint=0)
107
108 # If the output contains line breaks, try 2-arg show instead.
109 if occursin('\n', sx)
110 sx = sprint(show, x, context=io, sizehint=0)
111 end
112 else
113 a = undef_ref_alignment
114 sx = undef_ref_str
115 end
116 l = repeat(" ", A[k][1]-a[1]) # pad on left and right as needed
117 r = j == axes(X, 2)[end] ? "" : repeat(" ", A[k][2]-a[2])
118 prettysx = replace_in_print_matrix(X,i,j,sx)
119 print(io, l, prettysx, r)
120 if k < length(A); print(io, sep); end
121 end
122 end

125 """
126 `print_matrix_vdots` is used to show a series of vertical ellipsis instead
127 of a bunch of rows for long matrices. Not only is the string vdots shown
128 but it also repeated every M elements if desired.
129 """
130 function print_matrix_vdots(io::IO, vdots::AbstractString,
131 A::Vector, sep::AbstractString, M::Integer, m::Integer,
132 pad_right::Bool = true)
133 for k = 1:length(A)
134 w = A[k][1] + A[k][2]
135 if k % M == m
136 l = repeat(" ", max(0, A[k][1]-length(vdots)))
137 r = k == length(A) && !pad_right ?
138 "" :
139 repeat(" ", max(0, w-length(vdots)-length(l)))
140 print(io, l, vdots, r)
141 else
142 (k != length(A) || pad_right) && print(io, repeat(" ", w))
143 end
144 if k < length(A); print(io, sep); end
145 end
146 end

172 function _print_matrix(io, @nospecialize(X::AbstractVecOrMat), pre, sep, post, hdots, vdots, ddots, hmod, vmod, rowsA, colsA)
173 hmod, vmod = Int(hmod)::Int, Int(vmod)::Int
174 if !(get(io, :limit, false)::Bool)
175 screenheight = screenwidth = typemax(Int)
176 else
177 sz = displaysize(io)::Tuple{Int,Int}
178 screenheight, screenwidth = sz[1] - 4, sz[2]
179 end
180 screenwidth -= length(pre)::Int + length(post)::Int
181 presp = repeat(" ", length(pre)::Int) # indent each row to match pre string
182 postsp = ""
183 @assert textwidth(hdots) == textwidth(ddots)
184 sepsize = length(sep)::Int
185 m, n = length(rowsA), length(colsA)
186 # To figure out alignments, only need to look at as many rows as could
187 # fit down screen. If screen has at least as many rows as A, look at A.
188 # If not, then we only need to look at the first and last chunks of A,
189 # each half a screen height in size.
190 halfheight = div(screenheight,2)
191 if m > screenheight
192 rowsA = [rowsA[(0:halfheight-1) .+ firstindex(rowsA)]; rowsA[(end-div(screenheight-1,2)+1):end]]
193 else
194 rowsA = [rowsA;]
195 end
196 # Similarly for columns, only necessary to get alignments for as many
197 # columns as could conceivably fit across the screen
198 maxpossiblecols = div(screenwidth, 1+sepsize)
199 if n > maxpossiblecols
200 colsA = [colsA[(0:maxpossiblecols-1) .+ firstindex(colsA)]; colsA[(end-maxpossiblecols+1):end]]
201 else
202 colsA = [colsA;]
203 end
204 A = alignment(io, X, rowsA, colsA, screenwidth, screenwidth, sepsize)
205 # Nine-slicing is accomplished using print_matrix_row repeatedly
206 if m <= screenheight # rows fit vertically on screen
207 if n <= length(A) # rows and cols fit so just print whole matrix in one piece
208 for i in rowsA
209 print(io, i == first(rowsA) ? pre : presp)
210 print_matrix_row(io, X,A,i,colsA,sep)
211 print(io, i == last(rowsA) ? post : postsp)
212 if i != last(rowsA); println(io); end
213 end
214 else # rows fit down screen but cols don't, so need horizontal ellipsis
215 c = div(screenwidth-length(hdots)::Int+1,2)+1 # what goes to right of ellipsis
216 Ralign = reverse(alignment(io, X, rowsA, reverse(colsA), c, c, sepsize)) # alignments for right
217 c = screenwidth - sum(map(sum,Ralign)) - (length(Ralign)-1)*sepsize - length(hdots)::Int
218 Lalign = alignment(io, X, rowsA, colsA, c, c, sepsize) # alignments for left of ellipsis
219 for i in rowsA
220 print(io, i == first(rowsA) ? pre : presp)
221 print_matrix_row(io, X,Lalign,i,colsA[1:length(Lalign)],sep)
222 print(io, (i - first(rowsA)) % hmod == 0 ? hdots : repeat(" ", length(hdots)::Int))
223 print_matrix_row(io, X, Ralign, i, (n - length(Ralign)) .+ colsA, sep)
224 print(io, i == last(rowsA) ? post : postsp)
225 if i != last(rowsA); println(io); end
226 end
227 end
228 else # rows don't fit so will need vertical ellipsis
229 if n <= length(A) # rows don't fit, cols do, so only vertical ellipsis
230 for i in rowsA
231 print(io, i == first(rowsA) ? pre : presp)
232 print_matrix_row(io, X,A,i,colsA,sep)
233 print(io, i == last(rowsA) ? post : postsp)
234 if i != rowsA[end] || i == rowsA[halfheight]; println(io); end
235 if i == rowsA[halfheight]
236 print(io, i == first(rowsA) ? pre : presp)
237 print_matrix_vdots(io, vdots, A, sep, vmod, 1, false)
238 print(io, i == last(rowsA) ? post : postsp * '\n')
239 end
240 end
241 else # neither rows nor cols fit, so use all 3 kinds of dots
242 c = div(screenwidth-length(hdots)::Int+1,2)+1
243 Ralign = reverse(alignment(io, X, rowsA, reverse(colsA), c, c, sepsize))
244 c = screenwidth - sum(map(sum,Ralign)) - (length(Ralign)-1)*sepsize - length(hdots)::Int
245 Lalign = alignment(io, X, rowsA, colsA, c, c, sepsize)
246 r = mod((length(Ralign)-n+1),vmod) # where to put dots on right half
247 for i in rowsA
248 print(io, i == first(rowsA) ? pre : presp)
249 print_matrix_row(io, X,Lalign,i,colsA[1:length(Lalign)],sep)
250 print(io, (i - first(rowsA)) % hmod == 0 ? hdots : repeat(" ", length(hdots)::Int))
251 print_matrix_row(io, X,Ralign,i,(n-length(Ralign)).+colsA,sep)
252 print(io, i == last(rowsA) ? post : postsp)
253 if i != rowsA[end] || i == rowsA[halfheight]; println(io); end
254 if i == rowsA[halfheight]
255 print(io, i == first(rowsA) ? pre : presp)
256 print_matrix_vdots(io, vdots, Lalign, sep, vmod, 1, true)
257 print(io, ddots)
258 print_matrix_vdots(io, vdots, Ralign, sep, vmod, r, false)
259 print(io, i == last(rowsA) ? post : postsp * '\n')
260 end
261 end
262 end
263 if isempty(rowsA)
264 print(io, pre)
265 print(io, vdots)
266 length(colsA) > 1 && print(io, " ", ddots)
267 print(io, post)
268 end
269 end
