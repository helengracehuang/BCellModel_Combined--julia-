362 y = reinterpret(Float64,(zu & 0x000f_ffff_ffff_ffff) | 0x3ff0_0000_0000_0000)
363
364 mf = Float64(m)
365 F = (y + 3.5184372088832e13) - 3.5184372088832e13 # 0x1p-7*round(0x1p7*y)
366 f = (y - F) + c*s #2^m(F+f) = 1+x = z+c
367 jp = unsafe_trunc(Int,128.0*F)-127
368
369 log_proc1(y,mf,F,f,jp)
370 elseif x == -1.0
371 -Inf
372 elseif isnan(x)
373 NaN
374 else
375 throw_complex_domainerror(:log1p, x)
376 end
377 end

379 function log1p(x::Float32)
380 if x > -1f0
381 x == Inf32 && return x
382 if -5.9604645f-8 < x < 5.9604645f-8
383 return x # Inexact
384 # Step 2
385 elseif -0.06058694f0 < x < 0.06449446f0
386 return log_proc2(x)
387 end
388
389 # Step 3
390 z = 1f0 + x
391 zu = reinterpret(UInt32,z)
392 s = reinterpret(Float32,0x7f000000 - (zu & 0xff80_0000)) # 2^-m
393 m = Int(zu >> 23) & 0x00ff - 127 # z cannot be subnormal
394 c = m > 0 ? 1f0-(z-x) : x-(z-1f0) # 1+x = z+c
395 y = reinterpret(Float32,(zu & 0x007f_ffff) | 0x3f80_0000)
396
397 mf = Float32(m)
398 F = (y + 65536.0f0) - 65536.0f0 # 0x1p-7*round(0x1p7*y)
399 f = (y - F) + s*c #2^m(F+f) = 1+x = z+c
400 jp = unsafe_trunc(Int,128.0*F)-127
401
402 log_proc1(y,mf,F,f,jp)
403 elseif x == -1f0
404 -Inf32
405 elseif isnan(x)
406 NaN32
407 else
408 throw_complex_domainerror(:log1p, x)
409 end
410 end
