151 """
152 isinplace(f, inplace_param_number[,fname="f"])
153 isinplace(f::AbstractSciMLFunction[, inplace_param_number])
154
155 Check whether a function operates in place by comparing its number of arguments
156 to the expected number. If `f` is an `AbstractSciMLFunction`, then the type
157 parameter is assumed to be correct and is used. Otherwise `inplace_param_number`
158 is checked against the methods table, where `inplace_param_number` is the number
159 of arguments for the in-place dispatch. The out-of-place dispatch is assumed
160 to have one less. If neither of these dispatches exist, an error is thrown.
161 If the error is thrown, `fname` is used to tell the user which function has the
162 incorrect dispatches.
163
164 # See also
165 * [`numargs`](@ref numargs)
166 """
167 function isinplace(f,inplace_param_number,fname="f")
168 nargs = numargs(f)
169 iip_dispatch = any(x->x==inplace_param_number,nargs)
170 oop_dispatch = any(x->x==inplace_param_number-1,nargs)
171
172 if !iip_dispatch && !oop_dispatch
173 if length(nargs) == 0
174 throw(NoMethodsError(fname))
175 elseif all(x->x>inplace_param_number,nargs)
176 throw(TooManyArgumentsError(fname))
177 elseif all(x->x<inplace_param_number-1,nargs)
178 # Possible extra safety?
179 # Find if there's a `f(args...)` dispatch
180 # If so, no error
181 for i in 1:length(nargs)
182 if nargs[i] < inplace_param_number && any(isequal(Vararg{Any}),methods(f).ms[1].sig.parameters)
183 # If varargs, assume iip
184 return true
185 end
186 end
187
188 # No varargs detected, error that there are dispatches but not the right ones
189
190 throw(TooFewArgumentsError(fname))
191 else
192 throw(FunctionArgumentsError(fname))
193 end
194 else
195 # Equivalent to, if iip_dispatch exists, treat as iip
196 # Otherwise, it's oop
197 iip_dispatch
198 end
199 end

314 function add_kwonly(::Type{Val{:call}}, default_call::Expr)
315 # default_call is, e.g., :(f(a, b=2; c=3))
316 funcname = default_call.args[1] # e.g., :f
317 required = [] # required positional arguments; e.g., [:a]
318 optional = [] # optional positional arguments; e.g., [:(b=2)]
319 default_kwargs = []
320 for arg in default_call.args[2:end]
321 if isa(arg, Symbol)
322 push!(required, arg)
323 elseif arg.head == :(::)
324 push!(required, arg)
325 elseif arg.head == :kw
326 push!(optional, arg)
327 elseif arg.head == :parameters
328 @assert default_kwargs == [] # can I have :parameters twice?
329 default_kwargs = arg.args
330 else
331 error("Not expecting to see: $arg")
332 end
333 end
334 if isempty(required) && isempty(optional)
335 # If the function is already keyword-only, do nothing:
336 return nothing
337 end
338 if isempty(required)
339 # It's not clear what should be done. Let's not support it at
340 # the moment:
341 error("At least one positional mandatory argument is required.")
342 end
343
344 kwonly_kwargs = Expr(:parameters, [
345 Expr(:kw, pa, :(error($("No argument $pa"))))
346 for pa in required
347 ]..., optional..., default_kwargs...)
348 kwonly_call = Expr(:call, funcname, kwonly_kwargs)
349 # e.g., :(f(; a=error(...), b=error(...), c=1, d=2))
350
351 return kwonly_call
352 end
