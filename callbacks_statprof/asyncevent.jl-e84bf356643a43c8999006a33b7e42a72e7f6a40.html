38 """
39 AsyncCondition(callback::Function)
40
41 Create a async condition that calls the given `callback` function. The `callback` is passed one argument,
42 the async condition object itself.
43 """
44 function AsyncCondition(cb::Function)
45 async = AsyncCondition()
46 t = @task begin
47 unpreserve_handle(async)
48 while _trywait(async)
49 cb(async)
50 isopen(async) || return
51 end
52 end
53 # here we are mimicking parts of _trywait, in coordination with task `t`
54 preserve_handle(async)
55 @lock async.cond begin
56 if async.set
57 schedule(t)
58 else
59 _wait2(async.cond, t)
60 end
61 end
62 return async
63 end

65 ## timer-based notifications

67 """
68 Timer(delay; interval = 0)
69
70 Create a timer that wakes up tasks waiting for it (by calling [`wait`](@ref) on the timer object).
71
72 Waiting tasks are woken after an initial delay of at least `delay` seconds, and then repeating after
73 at least `interval` seconds again elapse. If `interval` is equal to `0`, the timer is only triggered
74 once. When the timer is closed (by [`close`](@ref)) waiting tasks are woken with an error. Use
75 [`isopen`](@ref) to check whether a timer is still active.
76
77 Note: `interval` is subject to accumulating time skew. If you need precise events at a particular
78 absolute time, create a new timer at each expiration with the difference to the next time computed.
79 """
80 mutable struct Timer
81 handle::Ptr{Cvoid}
82 cond::ThreadSynchronizer
83 isopen::Bool
84 set::Bool
85
86 function Timer(timeout::Real; interval::Real = 0.0)
87 timeout ≥ 0 || throw(ArgumentError("timer cannot have negative timeout of $timeout seconds"))
88 interval ≥ 0 || throw(ArgumentError("timer cannot have negative repeat interval of $interval seconds"))
89 # libuv has a tendency to timeout 1 ms early, so we need +1 on the timeout (in milliseconds), unless it is zero
90 timeoutms = ceil(UInt64, timeout * 1000) + !iszero(timeout)
91 intervalms = ceil(UInt64, interval * 1000)
92 loop = eventloop()
93
94 this = new(Libc.malloc(_sizeof_uv_timer), ThreadSynchronizer(), true, false)
95 associate_julia_struct(this.handle, this)
96 iolock_begin()
97 err = ccall(:uv_timer_init, Cint, (Ptr{Cvoid}, Ptr{Cvoid}), loop, this)
98 @assert err == 0
99 finalizer(uvfinalize, this)
100 ccall(:uv_update_time, Cvoid, (Ptr{Cvoid},), loop)
101 err = ccall(:uv_timer_start, Cint, (Ptr{Cvoid}, Ptr{Cvoid}, UInt64, UInt64),
102 this, @cfunction(uv_timercb, Cvoid, (Ptr{Cvoid},)),
103 timeoutms, intervalms)
104 @assert err == 0
105 iolock_end()
106 return this
107 end
108 end

113 function _trywait(t::Union{Timer, AsyncCondition})
114 set = t.set
115 if set
116 # full barrier now for AsyncCondition
117 t isa Timer || Core.Intrinsics.atomic_fence(:acquire_release)
118 else
119 t.isopen || return false
120 t.handle == C_NULL && return false
121 iolock_begin()
122 set = t.set
123 if !set
124 preserve_handle(t)
125 lock(t.cond)
126 try
127 set = t.set
128 if !set && t.isopen && t.handle != C_NULL
129 iolock_end()
130 set = wait(t.cond)
131 unlock(t.cond)
132 iolock_begin()
133 lock(t.cond)
134 end
135 finally
136 unlock(t.cond)
137 unpreserve_handle(t)
138 end
139 end
140 iolock_end()
141 end
142 t.set = false
143 return set
144 end

165 iolock_begin()
166 lock(t.cond)
167 try
168 if t.handle != C_NULL
169 disassociate_julia_struct(t.handle) # not going to call the usual close hooks

171 t.isopen = false

173 end
174 t.handle = C_NULL

236 # timer with repeated callback
237 """
238 Timer(callback::Function, delay; interval = 0)
239
240 Create a timer that runs the function `callback` at each timer expiration.
241
242 Waiting tasks are woken and the function `callback` is called after an initial delay of `delay`
243 seconds, and then repeating with the given `interval` in seconds. If `interval` is equal to `0`, the
244 callback is only run once. The function `callback` is called with a single argument, the timer
245 itself. Stop a timer by calling `close`. The `cb` may still be run one final time, if the timer has
246 already expired.
247
248 # Examples
249
250 Here the first number is printed after a delay of two seconds, then the following numbers are
251 printed quickly.
252
253 ```julia-repl
254 julia> begin
255 i = 0
256 cb(timer) = (global i += 1; println(i))
257 t = Timer(cb, 2, interval=0.2)
258 wait(t)
259 sleep(0.5)
260 close(t)
261 end
262 1
263 2
264 3
265 ```
266 """

268 timer = Timer(timeout, interval=interval)

270 unpreserve_handle(timer)

273 1 (0 %)
1 (0 %) samples spent in macro expansion
1 (100 %) (incl.) when called from #635 line 123
1 (100 %) samples spent calling send_stdout
cb(timer)
274 catch err

276 showerror(stderr, err, catch_backtrace())
277 return
278 end

280 end
281 end

294 """
295 timedwait(callback::Function, timeout::Real; pollint::Real=0.1)
296
297 Waits until `callback` returns `true` or `timeout` seconds have passed, whichever is earlier.
298 `callback` is polled every `pollint` seconds. The minimum value for `timeout` and `pollint`
299 is `0.001`, that is, 1 millisecond.
300
301 Returns :ok or :timed_out
302 """
303 function timedwait(testcb::Function, timeout::Real; pollint::Real=0.1)
304 pollint >= 1e-3 || throw(ArgumentError("pollint must be ≥ 1 millisecond"))
305 start = time_ns()
306 ns_timeout = 1e9 * timeout
307 done = Channel(1)
308 function timercb(aw)
309 try
310 if testcb()
311 put!(done, (:ok, nothing))
312 elseif (time_ns() - start) > ns_timeout
313 put!(done, (:timed_out, nothing))
314 end
315 catch e
316 put!(done, (:error, CapturedException(e, catch_backtrace())))
317 finally
318 isready(done) && close(aw)
319 end
320 nothing
321 end
322
323 try
324 testcb() && return :ok
325 catch e
326 throw(CapturedException(e, catch_backtrace()))
327 end
328
329 t = Timer(timercb, pollint, interval = pollint)
330 ret, e = fetch(done)
331 close(t)
332
333 ret === :error && throw(e)
334
335 return ret
336 end
