170 function DiffEqBase.prepare_alg(alg::Union{OrdinaryDiffEqAdaptiveImplicitAlgorithm{0,AD,FDT},
171 OrdinaryDiffEqImplicitAlgorithm{0,AD,FDT},
172 DAEAlgorithm{0,AD,FDT}},u0::AbstractArray{T},p,prob) where {AD,FDT,T}
173 alg isa OrdinaryDiffEqImplicitExtrapolationAlgorithm && return alg # remake fails, should get fixed
174
175 if alg.linsolve === nothing
176 if (prob.f isa ODEFunction && prob.f.f isa SciMLBase.AbstractDiffEqOperator)
177 linsolve = LinearSolve.defaultalg(prob.f.f,u0)
178 elseif (prob.f isa SplitFunction && prob.f.f1.f isa SciMLBase.AbstractDiffEqOperator)
179 linsolve = LinearSolve.defaultalg(prob.f.f1.f,u0)
180 if (linsolve === nothing) | (linsolve isa LinearSolve.AbstractFactorization)
181 msg = "Split ODE problem do not work with factorization linear solvers. Bug detailed in https://github.com/SciML/OrdinaryDiffEq.jl/pull/1643. Defaulting to linsolve=KrylovJL()"
182 @warn msg
183 linsolve = KrylovJL()
184 end
185 elseif prob isa ODEProblem && (prob.f.mass_matrix === nothing ||
186 (prob.f.mass_matrix !== nothing &&
187 !(typeof(prob.f.jac_prototype) <: SciMLBase.AbstractDiffEqOperator)))
188 linsolve = LinearSolve.defaultalg(prob.f.jac_prototype,u0)
189 else
190 # If mm is a sparse matrix and A is a DiffEqArrayOperator, then let linear
191 # solver choose things later
192 linsolve = nothing
193 end
194 else
195 linsolve = alg.linsolve
196 end
197
198 isbitstype(T) && sizeof(T) > 24 && return remake(alg, chunk_size=Val{1}(),linsolve=linsolve)
199
200 L = ArrayInterface.known_length(typeof(u0))
201 if L === nothing # dynamic sized
202
203 # If chunksize is zero, pick chunksize right at the start of solve and
204 # then do function barrier to infer the full solve
205 x = if prob.f.colorvec === nothing
206 length(u0)
207 else
208 maximum(prob.f.colorvec)
209 end
210
211 cs = ForwardDiff.pickchunksize(x)
212 remake(alg,chunk_size=Val{cs}(),linsolve=linsolve)
213 else # statically sized
214 cs = pick_static_chunksize(Val{L}())
215 remake(alg,chunk_size=cs,linsolve=linsolve)
216 end
217 end

219 @generated function pick_static_chunksize(::Val{chunksize}) where chunksize
220 x = ForwardDiff.pickchunksize(chunksize)
221 :(Val{$x}())
222 end

593 function default_controller(alg, cache, qoldinit, _beta1=nothing, _beta2=nothing)
594 if ispredictive(alg)
595 return PredictiveController()
596 elseif isstandard(alg)
597 return IController()
598 else # Default is PI-controller
599 QT = typeof(qoldinit)
600 beta1, beta2 = _digest_beta1_beta2(alg, cache, Val(QT), _beta1, _beta2)
601 return PIController(beta1, beta2)
602 end
603 end

605 function default_controller(alg::Union{ExtrapolationMidpointDeuflhard,ImplicitDeuflhardExtrapolation, ExtrapolationMidpointHairerWanner, ImplicitHairerWannerExtrapolation, ImplicitEulerExtrapolation, ImplicitEulerBarycentricExtrapolation}, cache, qoldinit, _beta1=nothing, _beta2=nothing)
606 QT = typeof(qoldinit)
607 beta1, beta2 = _digest_beta1_beta2(alg, cache, Val(QT), _beta1, _beta2)
608 return ExtrapolationController(beta1)
609 end
610
611 function _digest_beta1_beta2(alg, cache, ::Val{QT}, _beta1, _beta2) where {QT}
612 if typeof(alg) <: OrdinaryDiffEqCompositeAlgorithm
613 beta2 = _beta2 === nothing ? _composite_beta2_default(alg.algs, cache.current, Val(QT)) : _beta2
614 beta1 = _beta1 === nothing ? _composite_beta1_default(alg.algs, cache.current, Val(QT), beta2) : _beta1
615 else
616 beta2 = _beta2 === nothing ? beta2_default(alg) : _beta2
617 beta1 = _beta1 === nothing ? beta1_default(alg,beta2) : _beta1
618 end
619 return convert(QT, beta1)::QT, convert(QT, beta2)::QT
620 end

759 function unwrap_alg(alg::SciMLBase.DEAlgorithm, is_stiff)
760 iscomp = typeof(alg) <: CompositeAlgorithm
761 if !iscomp
762 return alg
763 elseif typeof(alg.choice_function) <: AutoSwitchCache
764 if is_stiff === nothing
765 throwautoswitch(alg)
766 end
767 num = is_stiff ? 2 : 1
768 if num == 1
769 return alg.algs[1]
770 else
771 return alg.algs[2]
772 end
773 else
774 error("this dispatch does not support this algorithm right now")
775 end
776 end

778 function unwrap_alg(integrator, is_stiff)
779 alg = integrator.alg
780 iscomp = typeof(alg) <: CompositeAlgorithm
781 if !iscomp
782 return alg
783 elseif typeof(alg.choice_function) <: AutoSwitchCache
784 if is_stiff === nothing
785 throwautoswitch(alg)
786 end
787 num = is_stiff ? 2 : 1
788 if num == 1
789 return alg.algs[1]
790 else
791 return alg.algs[2]
792 end
793 else
794 if integrator.cache.current == 1
795 return alg.algs[1]
796 elseif integrator.cache.current == 2
797 return alg.algs[2]
798 else
799 return alg.algs[integrator.cache.current]
800 end
801 end
802 end
