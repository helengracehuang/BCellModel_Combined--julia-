70 const max_output_per_request = Ref(1 << 19)

72 """
73 Continually read from (size limited) Libuv/OS buffer into an `IObuffer` to avoid problems when
74 the Libuv/OS buffer gets full (https://github.com/JuliaLang/julia/issues/8789). Send data immediately
75 when buffer contains more than `max_bytes` bytes. Otherwise, if data is available it will be sent every
76 `stream_interval` seconds (see the Timers set up in watch_stdio). Truncate the output to `max_output_per_request`
77 bytes per execution request since excessive output can bring browsers to a grinding halt.
78
1 (0 %) samples spent in watch_stream
1 (100 %) (incl.) when called from #23 line 429
"""
79 function watch_stream(rd::IO, name::AbstractString)
80 task_local_storage(:IJulia_task, "read $name task")
81 try
82 buf = IOBuffer()
83 bufs[name] = buf
84 1 (0 %)
1 (100 %) samples spent calling eof
while !eof(rd) # blocks until something is available
85 nb = bytesavailable(rd)
86 if nb > 0
87 stdio_bytes[] += nb
88 # if this stream has surpassed the maximum output limit then ignore future bytes
89 if stdio_bytes[] >= max_output_per_request[]
90 read(rd, nb) # read from libuv/os buffer and discard
91 if stdio_bytes[] - nb < max_output_per_request[]
92 send_ipython(publish[], msg_pub(execute_msg, "stream",
93 Dict("name" => "stderr", "text" => "Excessive output truncated after $(stdio_bytes[]) bytes.")))
94 end
95 else
96 write(buf, read(rd, nb))
97 end
98 end
99 if buf.size > 0
100 if buf.size >= max_bytes
101 #send immediately
102 send_stream(name)
103 end
104 end
105 end
106 catch e
107 # the IPython manager may send us a SIGINT if the user
108 # chooses to interrupt the kernel; don't crash on this
109 if isa(e, InterruptException)
110 watch_stream(rd, name)
111 else
112 rethrow()
113 end
114 end
115 end

127 """
128 Jupyter associates cells with message headers. Once a cell's execution state has
129 been set as to idle, it will silently drop stream messages (i.e. output to
130 stdout and stderr) - see https://github.com/jupyter/notebook/issues/518.
131 When using Interact, and a widget's state changes, a new
132 message header is sent to the IJulia kernel, and while Reactive
133 is updating Signal graph state, it's execution state is busy, meaning Jupyter
134 will not drop stream messages if Interact can set the header message under which
135 the stream messages will be sent. Hence the need for this function.
136 """

141 function send_stream(name::AbstractString)
142 buf = bufs[name]
143 if buf.size > 0
144 d = take!(buf)
145 n = num_utf8_trailing(d)
146 dextra = d[end-(n-1):end]
147 resize!(d, length(d) - n)
148 s = String(copy(d))
149 if isvalid(String, s)
150 write(buf, dextra) # assume that the rest of the string will be written later
151 length(d) == 0 && return
152 else
153 # fallback: base64-encode non-UTF8 binary data
154 sbuf = IOBuffer()
155 print(sbuf, "base64 binary data: ")
156 b64 = Base64EncodePipe(sbuf)
157 write(b64, d)
158 write(b64, dextra)
159 close(b64)
160 print(sbuf, '\n')
161 s = String(take!(sbuf))
162 end
163 send_ipython(publish[],
164 msg_pub(execute_msg, "stream",
165 Dict("name" => name, "text" => s)))
166 end
167 end

169 """
170 If `d` ends with an incomplete UTF8-encoded character, return the number of trailing incomplete bytes.
171 Otherwise, return `0`.
172 """
173 function num_utf8_trailing(d::Vector{UInt8})
174 i = length(d)
175 # find last non-continuation byte in d:
176 while i >= 1 && ((d[i] & 0xc0) == 0x80)
177 i -= 1
178 end
179 i < 1 && return 0
180 c = d[i]
181 # compute number of expected UTF-8 bytes starting at i:
182 n = c <= 0x7f ? 1 : c < 0xe0 ? 2 : c < 0xf0 ? 3 : 4
183 nend = length(d) + 1 - i # num bytes from i to end
184 return nend == n ? 0 : nend
185 end
