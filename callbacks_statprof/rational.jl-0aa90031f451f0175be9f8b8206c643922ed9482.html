157 function rationalize(::Type{T}, x::AbstractFloat, tol::Real) where T<:Integer

159 throw(ArgumentError("negative tolerance $tol"))
160 end

162 isnan(x) && return T(x)//one(T)
163 isinf(x) && return unsafe_rational(x < 0 ? -one(T) : one(T), zero(T))

165 p, q = (x < 0 ? -one(T) : one(T)), zero(T)
166 pp, qq = zero(T), one(T)
167
168 x = abs(x)
169 a = trunc(x)
170 r = x-a
171 y = one(x)
172
173 tolx = oftype(x, tol)
174 nt, t, tt = tolx, zero(tolx), tolx
175 ia = np = nq = zero(T)
176
177 # compute the successive convergents of the continued fraction
178 # np // nq = (p*a + pp) // (q*a + qq)
179 while r > nt
180 try
181 ia = convert(T,a)
182
183 np = checked_add(checked_mul(ia,p),pp)
184 nq = checked_add(checked_mul(ia,q),qq)
185 p, pp = np, p
186 q, qq = nq, q
187 catch e
188 isa(e,InexactError) || isa(e,OverflowError) || rethrow()
189 return p // q
190 end

192 # naive approach of using
193 # x = 1/r; a = trunc(x); r = x - a
194 # is inexact, so we store x as x/y
195 x, y = y, r
196 a, r = divrem(x,y)
197
198 # maintain
199 # x0 = (p + (-1)^i * r) / q
200 t, tt = nt, t
201 nt = a*t+tt

204 # find optimal semiconvergent
205 # smallest a such that x-a*y < a*t+tt
206 a = cld(x-tt,y+t)
