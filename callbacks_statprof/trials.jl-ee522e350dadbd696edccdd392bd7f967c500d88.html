215 function judge(ratio::Real, tolerance::Float64)
216 if isnan(ratio) || (ratio - tolerance) > 1.0
217 return :regression
218 elseif (ratio + tolerance) < 1.0
219 return :improvement
220 else
221 return :invariant
222 end
223 end

293 function bindata(sorteddata, nbins, min, max)
294 Δ = (max - min) / nbins
295 bins = zeros(nbins)
296 lastpos = 0
297 for i ∈ 1:nbins

299 bins[i] = pos - lastpos

301 end

306 function asciihist(bins, height=1)
307 histbars = ['▁', '▂', '▃', '▄', '▅', '▆', '▇', '█']
308 if minimum(bins) == 0
309 barheights = 2 .+ round.(Int, (height * length(histbars) - 2) * bins ./ maximum(bins))
310 barheights[bins .== 0] .= 1
311 else
312 barheights = 1 .+ round.(Int, (height * length(histbars) - 1) * bins ./ maximum(bins))
313 end
314 heightmatrix = [min(length(histbars), barheights[b] - (h-1) * length(histbars))
315 for h in height:-1:1, b in 1:length(bins)]
316 map(height -> if height < 1; ' ' else histbars[height] end, heightmatrix)
317 end
