3 # Support for @fastmath
4
5 # This module provides versions of math functions that may violate
6 # strict IEEE semantics.
7
8 # This allows the following transformations. For more information see
9 # http://llvm.org/docs/LangRef.html#fast-math-flags:
10 # nnan: No NaNs - Allow optimizations to assume the arguments and
11 # result are not NaN. Such optimizations are required to retain
12 # defined behavior over NaNs, but the value of the result is
13 # undefined.
14 # ninf: No Infs - Allow optimizations to assume the arguments and
15 # result are not +/-Inf. Such optimizations are required to
16 # retain defined behavior over +/-Inf, but the value of the
17 # result is undefined.
18 # nsz: No Signed Zeros - Allow optimizations to treat the sign of a
19 # zero argument or result as insignificant.
20 # arcp: Allow Reciprocal - Allow optimizations to use the reciprocal
21 # of an argument rather than perform division.
22 # fast: Fast - Allow algebraically equivalent transformations that may
23 # dramatically change results in floating point (e.g.
24 # reassociate). This flag implies all the others.
25

94 function make_fastmath(expr::Expr)
95 if expr.head === :quote
96 return expr
97 elseif expr.head === :call && expr.args[1] === :^ && expr.args[3] isa Integer
98 # mimic Julia's literal_pow lowering of literal integer powers
99 return Expr(:call, :(Base.FastMath.pow_fast), make_fastmath(expr.args[2]), Val{expr.args[3]}())
100 end
101 op = get(rewrite_op, expr.head, :nothing)
102 if op !== :nothing
103 var = expr.args[1]
104 rhs = expr.args[2]
105 if isa(var, Symbol)
106 # simple assignment
107 expr = :($var = $op($var, $rhs))
108 elseif isa(var, Expr) && var.head === :ref
109 # array reference
110 arr = var.args[1]
111 inds = var.args[2:end]
112 arrvar = gensym()
113 indvars = Any[gensym() for i in inds]
114 expr = quote
115 $(Expr(:(=), arrvar, arr))
116 $(Expr(:(=), Expr(:tuple, indvars...), Expr(:tuple, inds...)))
117 $arrvar[$(indvars...)] = $op($arrvar[$(indvars...)], $rhs)
118 end
119 end
120 end
121 Expr(make_fastmath(expr.head), map(make_fastmath, expr.args)...)
122 end
123 function make_fastmath(symb::Symbol)
124 fast_symb = get(fast_op, symb, :nothing)
125 if fast_symb === :nothing
126 return symb
127 end
128 :(Base.FastMath.$fast_symb)
129 end

132 """
133 @fastmath expr
134
135 Execute a transformed version of the expression, which calls functions that
136 may violate strict IEEE semantics. This allows the fastest possible operation,
137 but results are undefined -- be careful when doing this, as it may change numerical
138 results.
139
140 This sets the [LLVM Fast-Math flags](http://llvm.org/docs/LangRef.html#fast-math-flags),
141 and corresponds to the `-ffast-math` option in clang. See [the notes on performance
142 annotations](@ref man-performance-annotations) for more details.
143
144 # Examples
145 ```jldoctest
146 julia> @fastmath 1+2
147 3
148
149 julia> @fastmath(sin(3))
150 0.1411200080598672
151 ```
152 """
153 macro fastmath(expr)
154 make_fastmath(esc(expr))
155 end
