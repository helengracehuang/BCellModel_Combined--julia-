3 """
4 replace_ref_begin_end!(ex)
5
6 Recursively replace occurrences of the symbols `:begin` and `:end` in a "ref" expression
7 (i.e. `A[...]`) `ex` with the appropriate function calls (`firstindex` or `lastindex`).
8 Replacement uses the closest enclosing ref, so
9
10 A[B[end]]
11
12 should transform to
13
14 A[B[lastindex(B)]]
15
16 """

18 # replace_ref_begin_end_!(ex,withex) returns (new ex, whether withex was used)
19 function replace_ref_begin_end_!(ex, withex)
20 used_withex = false
21 if isa(ex,Symbol)
22 if ex === :begin
23 withex === nothing && error("Invalid use of begin")
24 return withex[1], true
25 elseif ex === :end
26 withex === nothing && error("Invalid use of end")
27 return withex[2], true
28 end
29 elseif isa(ex,Expr)
30 if ex.head === :ref
31 ex.args[1], used_withex = replace_ref_begin_end_!(ex.args[1], withex)
32 S = isa(ex.args[1],Symbol) ? ex.args[1]::Symbol : gensym(:S) # temp var to cache ex.args[1] if needed
33 used_S = false # whether we actually need S
34 # new :ref, so redefine withex
35 nargs = length(ex.args)-1
36 if nargs == 0
37 return ex, used_withex
38 elseif nargs == 1
39 # replace with lastindex(S)
40 ex.args[2], used_S = replace_ref_begin_end_!(ex.args[2], (:($firstindex($S)),:($lastindex($S))))
41 else
42 n = 1
43 J = lastindex(ex.args)
44 for j = 2:J
45 exj, used = replace_ref_begin_end_!(ex.args[j], (:($firstindex($S)),:($lastindex($S,$n))))
46 used_S |= used
47 ex.args[j] = exj
48 if isa(exj,Expr) && exj.head === :...
49 # splatted object
50 exjs = exj.args[1]
51 n = :($n + length($exjs))
52 elseif isa(n, Expr)
53 # previous expression splatted
54 n = :($n + 1)
55 else
56 # an integer
57 n += 1
58 end
59 end
60 end
61 if used_S && S !== ex.args[1]
62 S0 = ex.args[1]
63 ex.args[1] = S
64 ex = Expr(:let, :($S = $S0), ex)
65 end
66 else
67 # recursive search
68 for i = eachindex(ex.args)
69 ex.args[i], used = replace_ref_begin_end_!(ex.args[i], withex)
70 used_withex |= used
71 end
72 end
73 end
74 ex, used_withex
75 end

77 """
78 @view A[inds...]
79
80 Creates a `SubArray` from an indexing expression. This can only be applied directly to a
81 reference expression (e.g. `@view A[1,2:end]`), and should *not* be used as the target of
82 an assignment (e.g. `@view(A[1,2:end]) = ...`). See also [`@views`](@ref)
83 to switch an entire block of code to use views for slicing.
84
85 !!! compat "Julia 1.5"
86 Using `begin` in an indexing expression to refer to the first index requires at least
87 Julia 1.5.
88
89 # Examples
90 ```jldoctest
91 julia> A = [1 2; 3 4]
92 2×2 Array{Int64,2}:
93 1 2
94 3 4
95
96 julia> b = @view A[:, 1]
97 2-element view(::Array{Int64,2}, :, 1) with eltype Int64:
98 1
99 3
100
101 julia> fill!(b, 0)
102 2-element view(::Array{Int64,2}, :, 1) with eltype Int64:
103 0
104 0
105
106 julia> A
107 2×2 Array{Int64,2}:
108 0 2
109 0 4
110 ```
111 """
112 macro view(ex)
113 if Meta.isexpr(ex, :ref)
114 ex = replace_ref_begin_end!(ex)
115 if Meta.isexpr(ex, :ref)
116 ex = Expr(:call, view, ex.args...)
117 else # ex replaced by let ...; foo[...]; end
118 @assert Meta.isexpr(ex, :let) && Meta.isexpr(ex.args[2], :ref)
119 ex.args[2] = Expr(:call, view, ex.args[2].args...)
120 end
121 Expr(:&&, true, esc(ex))
122 else
123 throw(ArgumentError("Invalid use of @view macro: argument must be a reference expression A[...]."))
124 end
125 end

138 # _views implements the transformation for the @views macro.
139 # @views calls esc(_views(...)) to work around #20241,
140 # so any function calls we insert (to maybeview, or to
141 # firstindex and lastindex in replace_ref_begin_end!) must be interpolated
142 # as values rather than as symbols to ensure that they are called
143 # from Base rather than from the caller's scope.
144 _views(x) = x
145 function _views(ex::Expr)
146 if ex.head in (:(=), :(.=))
147 # don't use view for ref on the lhs of an assignment,
148 # but still use views for the args of the ref:
149 lhs = ex.args[1]
150 Expr(ex.head, Meta.isexpr(lhs, :ref) ?
151 Expr(:ref, _views.(lhs.args)...) : _views(lhs),
152 _views(ex.args[2]))
153 elseif ex.head === :ref
154 Expr(:call, maybeview, _views.(ex.args)...)
155 else
156 h = string(ex.head)
157 # don't use view on the lhs of an op-assignment a[i...] += ...
158 if last(h) == '=' && Meta.isexpr(ex.args[1], :ref)
159 lhs = ex.args[1]
160
161 # temp vars to avoid recomputing a and i,
162 # which will be assigned in a let block:
163 a = gensym(:a)
164 i = [gensym(:i) for k = 1:length(lhs.args)-1]
165
166 # for splatted indices like a[i, j...], we need to
167 # splat the corresponding temp var.
168 I = similar(i, Any)
169 for k = 1:length(i)
170 if Meta.isexpr(lhs.args[k+1], :...)
171 I[k] = Expr(:..., i[k])
172 lhs.args[k+1] = lhs.args[k+1].args[1] # unsplat
173 else
174 I[k] = i[k]
175 end
176 end
177
178 Expr(:let,
179 Expr(:block,
180 :($a = $(_views(lhs.args[1]))),
181 [:($(i[k]) = $(_views(lhs.args[k+1]))) for k=1:length(i)]...),
182 Expr(first(h) == '.' ? :(.=) : :(=), :($a[$(I...)]),
183 Expr(:call, Symbol(h[1:end-1]),
184 :($maybeview($a, $(I...))),
185 _views.(ex.args[2:end])...)))
186 else
187 Expr(ex.head, _views.(ex.args)...)
188 end
189 end
190 end

192 """
193 @views expression
194
195 Convert every array-slicing operation in the given expression
196 (which may be a `begin`/`end` block, loop, function, etc.)
197 to return a view. Scalar indices, non-array types, and
198 explicit [`getindex`](@ref) calls (as opposed to `array[...]`) are
199 unaffected.
200
201 !!! note
202 The `@views` macro only affects `array[...]` expressions
203 that appear explicitly in the given `expression`, not array slicing that
204 occurs in functions called by that code.
205
206 !!! compat "Julia 1.5"
207 Using `begin` in an indexing expression to refer to the first index requires at least
208 Julia 1.5.
209
210 # Examples
211 ```jldoctest
212 julia> A = zeros(3, 3);
213
214 julia> @views for row in 1:3
215 b = A[row, :]
216 b[:] .= row
217 end
218
219 julia> A
220 3×3 Array{Float64,2}:
221 1.0 1.0 1.0
222 2.0 2.0 2.0
223 3.0 3.0 3.0
224 ```
225 """
